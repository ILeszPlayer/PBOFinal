    <!-- JAVASCRIPT LOGIC -->
    <script>
        // 1. Fitur Live Search
        function filterUsers() {
            let input = document.getElementById('searchInput').value.toLowerCase();
            let items = document.querySelectorAll('.user-item');
            let hasVisibleItem = false;

            items.forEach(item => {
                // Ambil data-name yang sudah kita set di HTML Thymeleaf
                let name = item.getAttribute('data-name'); 
                if (name.includes(input)) {
                    item.style.display = 'flex';
                    hasVisibleItem = true;
                } else {
                    item.style.display = 'none';
                }
            });

            // Tampilkan pesan "Tidak ditemukan" jika semua item ter-filter
            document.getElementById('noResult').style.display = (!hasVisibleItem && items.length > 0) ? 'block' : 'none';
        }

        // 2. Fitur Loading di Tombol Submit
        function showLoading(form) {
            const btnIcon = document.getElementById('btnIcon');
            const btnText = document.getElementById('btnText');
            const submitBtn = document.getElementById('submitBtn');
            
            // Ubah icon jadi spinner
            btnIcon.className = 'fas fa-circle-notch fa-spin';
            btnText.innerText = 'Memproses...';
            
            // Disable tombol agar tidak double click
            submitBtn.classList.add('opacity-80', 'cursor-not-allowed');
            submitBtn.disabled = true;
            
            // Lanjutkan submit form secara manual karena tombol di-disable
            form.submit(); 
        }

        // 3. Fitur Custom Modal Hapus
        const modal = document.getElementById('deleteModal');
        const modalContent = document.getElementById('modalContent');
        const confirmLink = document.getElementById('confirmDeleteLink');

        function openModal(deleteUrl) {
            // Set URL hapus ke tombol "Ya, Hapus"
            confirmLink.href = deleteUrl;
            
            // Tampilkan modal dengan animasi class
            modal.classList.add('modal-active');
            modalContent.classList.add('modal-content-active');
        }

        function closeModal() {
            // Hilangkan class animasi
            modal.classList.remove('modal-active');
            modalContent.classList.remove('modal-content-active');
            
            // Kosongkan kembali link
            setTimeout(() => confirmLink.href = "#", 300);
        }

        // Tutup modal jika klik area luar modal
        modal.addEventListener('click', function(e) {
            if (e.target === modal) closeModal();
        });
    </script>
